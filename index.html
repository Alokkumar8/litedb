Simple JSON Database for JavaScript

No signup, no login, no configuration. Just start storing and retrieving JSON data with simple HTTP requests.

⚡ Instant Setup
❌ No signup

Get Started in Seconds

1. Add to your HTML

<script src="https://litedb.code/cdn/script.js"></script>

2. Store Data (POST)

POST https://litedb.code/db/example-4343-erjknf

// Request Body
{
  "name": "Alok",
  "city": "Chennai"
}

3. Retrieve Data (GET)

GET https://litedb.code/db/example-4343-erjknf

// Response
{
  "name": "Alok",
  "city": "Chennai"
}

JavaScript Example

const dbId = 'example-4343-erjknf'
const url = `https://litedb.code/db/${dbId}`

// Store data
await fetch(url, {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ name: 'Alok', city: 'Chennai' })
})

// Retrieve data
const response = await fetch(url)
const data = await response.json()

HTML Form Submission

<!-- HTML -->
<form id="userForm">
  <input type="text" id="name" placeholder="Enter name" />
  <input type="text" id="city" placeholder="Enter city" />
  <button type="submit">Save Data</button>
</form>

<!-- JavaScript -->
<script>
document.getElementById('userForm').addEventListener('submit', async (e) => {
  e.preventDefault()
  const name = document.getElementById('name').value
  const city = document.getElementById('city').value
  await fetch('https://litedb.code/db/example-4343-erjknf', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ name, city })
  })
  alert('Data saved!')
})
</script>

Display Data on Webpage

<!-- HTML -->
<div id="userData">Loading...</div>
<button onclick="loadData()">Load Data</button>

<!-- JavaScript -->
<script>
async function loadData() {
  try {
    const response = await fetch('https://litedb.code/db/example-4343-erjknf')
    const data = await response.json()
    const userDiv = document.getElementById('userData')
    userDiv.innerHTML = `
      <h3>User Information</h3>
      <p><strong>Name:</strong> ${data.name}</p>
      <p><strong>City:</strong> ${data.city}</p>
    `
  } catch (error) {
    document.getElementById('userData').innerHTML = 'Error loading data'
  }
}

// Load data when page loads
loadData()
</script>
